    2. Run comprehensive verification queries:

    a. **Duplicate check** (tournament games):
    ```python
    import duckdb
    dups = duckdb.sql("""
        SELECT Season, DayNum, WTeamID, LTeamID, COUNT(*) as cnt
        FROM read_parquet('data/processed/tournament_games.parquet')
        GROUP BY ALL
        HAVING cnt > 1
    """).df()
    assert len(dups) == 0, f"Duplicates found: {dups}"
    ```

    b. **Season coverage check**:
    ```python
    seasons = duckdb.sql("""
        SELECT DISTINCT Season
        FROM read_parquet('data/processed/tournament_games.parquet')
        ORDER BY Season
    """).df()
    expected = [y for y in range(2003, 2026) if y != 2020]
    actual = sorted(seasons['Season'].tolist())
    assert actual == expected, f"Season mismatch: expected {expected}, got {actual}"
    ```

    c. **First Four accuracy check**:
    ```python
    ff = duckdb.sql("""
        SELECT Season, Round, COUNT(*) as games
        FROM read_parquet('data/processed/tournament_games.parquet')
        WHERE DayNum IN (134, 135)
        GROUP BY Season, Round
        ORDER BY Season
    """).df()
    # Pre-2011: should have 'Play-In Game' with 1 game per year
    # 2011+: should have 'First Four' with 4 games per year
    for _, row in ff.iterrows():
        if row['Season'] < 2011:
            assert row['Round'] == 'Play-In Game', f"Season {row['Season']} should be Play-In Game"
            assert row['games'] == 1, f"Season {row['Season']} should have 1 play-in game"
        else:
            assert row['Round'] == 'First Four', f"Season {row['Season']} should be First Four"
            assert row['games'] == 4, f"Season {row['Season']} should have 4 First Four games"
    ```

    d. **Cutoff enforcement test** (most critical verification):
    ```python
    from src.utils.query_helpers import get_season_stats_with_cutoff
    from src.utils.cutoff_dates import get_cutoff

    # For 2025 season, verify NO games after Selection Sunday are included
    stats_2025 = get_season_stats_with_cutoff(2025)
    cutoff_2025 = get_cutoff(2025)  # '2025-03-16'
    max_date = stats_2025['GameDate'].max()
    assert str(max_date)[:10] <= cutoff_2025, f"Cutoff violation: max game date {max_date} > {cutoff_2025}"
    print(f"2025 cutoff enforced: latest game {max_date}, cutoff {cutoff_2025}")
    ```

    e. **Normalization join check**:
    ```python
    unjoined = duckdb.sql("""
        SELECT DISTINCT t.WTeamID
        FROM read_parquet('data/processed/tournament_games.parquet') t

            ON t.WTeamID = n.kaggle_team_id

        WHERE n.kaggle_team_id IS NULL
    ```
