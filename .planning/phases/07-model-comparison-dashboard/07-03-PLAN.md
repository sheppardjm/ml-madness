```python
def select_best_model(
    models_data: dict[str, dict],
    output_path: str = "models/selected.json",
) -> dict:
    """Select best model by lowest mean Brier score and write selection artifact.

    Includes XGB and LGB aggregate Brier scores (from Phase 6 evaluation)
    alongside the full-backtest baseline and ensemble models.

    The output artifact is consumed by Phase 9 (Streamlit app) to determine
    which model to load for 2026 bracket predictions.

    Args:
        models_data: Dict from load_comparison_data() with baseline/ensemble results.
        output_path: Path to write the selection artifact JSON.

    Returns:
        The selection artifact dict.
    """
    # Include XGB and LGB aggregate Brier scores from Phase 6 evaluations
    # These models don't have full bracket-simulation backtest data, but we
    # know their mean Brier from temporal CV evaluation
    all_brier_scores = {
        "baseline": models_data["baseline"]["mean_brier"],
        "xgb": 0.1908,   # Phase 6 evaluate_xgb() result (06-01-SUMMARY)
        "lgb": 0.1931,   # Phase 6 evaluate_lgb() result (06-02-SUMMARY)
        "ensemble": models_data["ensemble"]["mean_brier"],
    }

    # Select best by lowest mean Brier
    best_name = min(all_brier_scores, key=all_brier_scores.get)
    best_brier = all_brier_scores[best_name]

    # Map model name to artifact path and type for Phase 9 model loading
    model_artifacts = {
        "baseline": {
            "model_artifact_path": "models/logistic_baseline.joblib",
            "model_type": "LogisticRegression",
        },
        "ensemble": {
            "model_artifact_path": "models/ensemble.joblib",
            "model_type": "TwoTierEnsemble",
        },
        # XGB and LGB don't have standalone bracket-simulation artifacts
        # (they are components of the ensemble), so they point to their
        # param files instead
        "xgb": {
            "model_artifact_path": "models/xgb_params.json",
            "model_type": "XGBClassifier",
        },
        "lgb": {
            "model_artifact_path": "models/lgb_params.json",
            "model_type": "LGBMClassifier",
        },
    }

    artifact = {
        "selected_model": best_name,
        "selection_criterion": "lowest_mean_brier",
        "mean_brier": round(best_brier, 4),
        "brier_scores": {k: round(v, 4) for k, v in all_brier_scores.items()},
        "model_artifact_path": model_artifacts[best_name]["model_artifact_path"],
        "model_type": model_artifacts[best_name]["model_type"],
        "evaluation_years": [2022, 2023, 2024, 2025],
        "notes": {
            "xgb_lgb_brier": "XGB and LGB Brier scores are from temporal CV evaluation (matchup-level), not full bracket simulation. They are included for ranking but lack per-round accuracy data.",
            "upset_tradeoff": f"Ensemble has significantly lower upset detection rate than baseline (~34% vs ~64% avg). Selected model ({best_name}) optimizes for Brier score, not upset detection.",
        },
        "generated_at": str(date.today()),
    }

    out = pathlib.Path(output_path)
    out.parent.mkdir(parents=True, exist_ok=True)
    out.write_text(json.dumps(artifact, indent=2))

    print()
    print(f"Selected model: {best_name} (mean Brier = {best_brier:.4f})")
    print(f"  Brier ranking: {' < '.join(f'{k}={v:.4f}' for k, v in sorted(all_brier_scores.items(), key=lambda x: x[1]))}")
    print(f"  Artifact: {model_artifacts[best_name]['model_artifact_path']}")
    print(f"  Selection written to: {output_path}")

    return artifact
```
